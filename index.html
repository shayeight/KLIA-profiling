        function renderMap() {

                    const to = document.getElementById(`node-${t.id}`);

                    if(from && to) {

                        const fromRect = from.getBoundingClientRect();

                        const toRect = to.getBoundingClientRect();

                        const svgRect = svg.getBoundingClientRect();

                        const x1 = (fromRect.left + fromRect.width / 2) - svgRect.left;

                        const y1 = (fromRect.top + fromRect.height) - svgRect.top;

                        const x2 = (toRect.left + toRect.width / 2) - svgRect.left;

                        const y2 = (toRect.top) - svgRect.top;

                        const path = document.createElementNS("http://www.w3.org/2000/svg", "path");

                        path.setAttribute("d", `M ${x1} ${y1} C ${x1} ${y1 + 60}, ${x2} ${y2 - 60}, ${x2} ${y2}`);

                        path.setAttribute("class", "line-connection"); svg.appendChild(path);

                    }

                }

            });

        }

        function renderAll() {

            const select = document.getElementById('syndicateFilter');

            const tags = document.getElementById('syndicateTags');

            const current = select.value;

            const uniqueSyns = [...new Set(targets.map(t => t.syndicateName))].sort();

           

            select.innerHTML = '<option value="">SEMUA SINDIKET</option>';

            tags.innerHTML = `<button onclick="setFilter('')" class="px-5 py-2 rounded-full text-[10px] font-black uppercase whitespace-nowrap transition-all ${current === '' ? 'bg-blue-600 text-white shadow-xl' : 'bg-white text-slate-400 border border-slate-200'}">Semua</button>`;

           

            uniqueSyns.forEach(s => {

                if(!s) return;

                const opt = document.createElement('option'); opt.value = s; opt.textContent = s; if(current === s) opt.selected = true; select.appendChild(opt);

                tags.innerHTML += `<button onclick="setFilter('${s}')" class="px-5 py-2 rounded-full text-[10px] font-black uppercase whitespace-nowrap transition-all ${current === s ? 'bg-blue-600 text-white shadow-xl' : 'bg-white text-slate-400 border border-slate-200'}">${s}</button>`;

            });

            if(isMapView) renderMap(); else renderList();

        }

        function setFilter(v) { document.getElementById('syndicateFilter').value = v; renderAll(); }

       

        // --- INITIAL START ---

        renderAll();

        setInterval(renderList, 60000); // Auto-refresh for flight alerts

        window.addEventListener('resize', () => { if(isMapView) drawMapLines(); });
